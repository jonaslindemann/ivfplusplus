INCLUDE_DIRECTORIES(${ivf_SOURCE_DIR}/include)
FILE(GLOB SOURCE_FILES *.cpp)
FILE(GLOB INCLUDE_FILES ${ivf_SOURCE_DIR}/include/ivfwin32/*.h)

IF(BUILD_SHARED_LIBS)
	ADD_DEFINITIONS(-DIVF_DLL -DIVFWIN32_LIB)
ENDIF(BUILD_SHARED_LIBS)

IF(WIN32)
	IF(IVFDEBUG)
		ADD_LIBRARY(ivfwin32d ${INCLUDE_FILES} ${SOURCE_FILES})
		TARGET_LINK_LIBRARIES( ivfwin32d ivfd ivfmathd ivfwidgetd )
		INSTALL_TARGETS( /lib ivfwin32d )
	ELSE(IVFDEBUG)
		ADD_LIBRARY(ivfwin32 ${INCLUDE_FILES} ${SOURCE_FILES})
		TARGET_LINK_LIBRARIES( ivfwin32 ivf ivfmath ivfwidget )
		INSTALL_TARGETS( /lib ivfwin32 )
	ENDIF(IVFDEBUG)
ENDIF(WIN32)
