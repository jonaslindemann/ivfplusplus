INCLUDE_DIRECTORIES(${ivf_SOURCE_DIR}/include)
FILE(GLOB SOURCE_FILES *.cpp)
FILE(GLOB INCLUDE_FILES ${ivf_SOURCE_DIR}/include/WIN32/*.h)

IF(BUILD_SHARED_LIBS)
	ADD_DEFINITIONS(-DIVF_DLL -DWIN32_LIB)
ENDIF(BUILD_SHARED_LIBS)

IF(WIN32)
	IF(IVFDEBUG)
		ADD_LIBRARY(WIN32d ${INCLUDE_FILES} ${SOURCE_FILES})
		TARGET_LINK_LIBRARIES( WIN32d ivfd ivfmathd ivfwidgetd )
		INSTALL_TARGETS( /lib WIN32d )
	ELSE(IVFDEBUG)
		ADD_LIBRARY(WIN32 ${INCLUDE_FILES} ${SOURCE_FILES})
		TARGET_LINK_LIBRARIES( WIN32 ivf ivfmath ivfwidget )
		INSTALL_TARGETS( /lib WIN32 )
	ENDIF(IVFDEBUG)
ENDIF(WIN32)
