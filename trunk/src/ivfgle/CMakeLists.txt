INCLUDE_DIRECTORIES(${ivf_SOURCE_DIR}/include)
FILE(GLOB SOURCE_FILES *.cpp)
FILE(GLOB INCLUDE_FILES ${ivf_SOURCE_DIR}/include/ivfgle/*.h)

IF(BUILD_SHARED_LIBS)
ADD_DEFINITIONS(-DIVF_DLL -DIVFGLE_LIB)
ENDIF(BUILD_SHARED_LIBS)

IF(WIN32)
	IF(IVFDEBUG)
		ADD_LIBRARY(ivfgled ${INCLUDE_FILES} ${SOURCE_FILES})
		TARGET_LINK_LIBRARIES( ivfgled ivfd ivfmathd opengl32 glu32 gled)
		INSTALL_TARGETS( /lib ivfgled )
	ELSE(IVFDEBUG)
		ADD_LIBRARY(ivfgle ${INCLUDE_FILES} ${SOURCE_FILES})
		TARGET_LINK_LIBRARIES( ivfgle ivf ivfmath opengl32 glu32 gle)
		INSTALL_TARGETS( /lib ivfgle )
	ENDIF(IVFDEBUG)		
ELSE(WIN32)
	ADD_LIBRARY(ivfgle ${SOURCE_FILES})
        IF(APPLE)
                TARGET_LINK_LIBRARIES( ivfgle ivf ivfmath "-framework Carbon" "-framework Cocoa" "-framework ApplicationServices" "-framework AGL" "-framework OpenGL" gle)
        ELSE(APPLE)
                TARGET_LINK_LIBRARIES( ivfgle ivf ivfmath GL GLU gle )
        ENDIF(APPLE)	
	INSTALL_TARGETS( /lib ivfgle )
ENDIF(WIN32)
