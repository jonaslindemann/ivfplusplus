INCLUDE_DIRECTORIES(${ivf_SOURCE_DIR}/include)
FILE(GLOB SOURCE_FILES *.cpp)
FILE(GLOB INCLUDE_FILES ${ivf_SOURCE_DIR}/include/ivffile/*.h)

IF(BUILD_SHARED_LIBS)
	ADD_DEFINITIONS(-DIVF_DLL -DIVFFILE_LIB)
ENDIF(BUILD_SHARED_LIBS)

IF(IVFWIN32)
	IF(IVFDEBUG)
		ADD_LIBRARY(ivffiled ${INCLUDE_FILES} ${SOURCE_FILES})
		TARGET_LINK_LIBRARIES( ivffiled ivfd ivfimaged )
		INSTALL_TARGETS( /lib ivffiled )
	ELSE(IVFDEBUG)
		ADD_LIBRARY(ivffile ${INCLUDE_FILES} ${SOURCE_FILES})
		TARGET_LINK_LIBRARIES( ivffile ivf ivfimage )
		INSTALL_TARGETS( /lib ivffile )
	ENDIF(IVFDEBUG)
ELSE(IVFWIN32)
	ADD_LIBRARY(ivffile ${SOURCE_FILES})
	TARGET_LINK_LIBRARIES( ivffile ivf ivfimage )
	INSTALL_TARGETS( /lib ivffile )
ENDIF(IVFWIN32)
